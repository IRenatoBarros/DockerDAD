db.contas.aggregate(
    {$lookup:{
        from:"clientes",
        localField:"id_cliente",
        foreignField:"_id",
        as:"clientes"
    }},
    {$unwind:"$clientes"},
    {$project:{
        _id:0,
        saldo:{
            $concat:["R$", 
            { $arrayElemAt:
            [{$split:[{$toString:"$valor"},"."]}]
            
            }]
        }
    }}
)
